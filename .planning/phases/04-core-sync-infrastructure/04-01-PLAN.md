    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    csdn_user_id VARCHAR(100) NOT NULL,
    category_id BIGINT,
    enabled BOOLEAN DEFAULT TRUE,
    last_sync_at TIMESTAMP NULL,

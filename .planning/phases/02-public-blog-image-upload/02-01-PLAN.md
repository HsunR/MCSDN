**findPublishedPaginated** - List published articles with pagination:
```xml
<select id="findPublishedPaginated" resultMap="ArticleResultMap">
  SELECT a.id, a.title, a.content, a.status, a.category_id, a.created_at, a.updated_at,
         c.name as category_name, c.id as cat_id
  FROM articles a
  LEFT JOIN categories c ON a.category_id = c.id
  WHERE a.status = 'PUBLISHED'
  ORDER BY a.created_at DESC
  LIMIT #{limit} OFFSET #{offset}
</select>
```

**countPublished** - Count total published articles for pagination:
```xml
<select id="countPublished" resultType="int">
  SELECT COUNT(*) FROM articles WHERE status = 'PUBLISHED'
</select>
```

**findByCategorySlugPaginated** - Filter by category (D-04: category slug matches via case-insensitive name query):
```xml
<select id="findByCategorySlugPaginated" resultMap="ArticleResultMap">
  SELECT a.id, a.title, a.content, a.status, a.category_id, a.created_at, a.updated_at,
         c.name as category_name, c.id as cat_id
  FROM articles a
  INNER JOIN categories c ON a.category_id = c.id
  WHERE a.status = 'PUBLISHED' AND LOWER(c.name) = LOWER(#{categorySlug})
  ORDER BY a.created_at DESC
  LIMIT #{limit} OFFSET #{offset}
</select>
```

**countByCategorySlug** - Count for category filter pagination:
```xml
<select id="countByCategorySlug" resultType="int">
  SELECT COUNT(*) FROM articles a
  INNER JOIN categories c ON a.category_id = c.id
  WHERE a.status = 'PUBLISHED' AND LOWER(c.name) = LOWER(#{categorySlug})
</select>
```

**findByTagSlugPaginated** - Filter by tag (D-05: tag slug via tag name):
```xml
<select id="findByTagSlugPaginated" resultMap="ArticleResultMap">
  SELECT DISTINCT a.id, a.title, a.content, a.status, a.category_id, a.created_at, a.updated_at,
         c.name as category_name, c.id as cat_id
  FROM articles a
  INNER JOIN article_tags at ON a.id = at.article_id
  INNER JOIN tags t ON at.tag_id = t.id
  LEFT JOIN categories c ON a.category_id = c.id
  WHERE a.status = 'PUBLISHED' AND LOWER(t.name) = LOWER(#{tagSlug})
  ORDER BY a.created_at DESC
  LIMIT #{limit} OFFSET #{offset}
</select>
```

**countByTagSlug** - Count for tag filter pagination:
```xml
<select id="countByTagSlug" resultType="int">
  SELECT COUNT(DISTINCT a.id) FROM articles a
  INNER JOIN article_tags at ON a.id = at.article_id
  INNER JOIN tags t ON at.tag_id = t.id
  WHERE a.status = 'PUBLISHED' AND LOWER(t.name) = LOWER(#{tagSlug})
</select>
```

**searchPublishedPaginated** - MySQL FULLTEXT search (D-07: MATCH() AGAINST() in NATURAL LANGUAGE MODE):
```xml
<select id="searchPublishedPaginated" resultMap="ArticleResultMap">
  SELECT a.id, a.title, a.content, a.status, a.category_id, a.created_at, a.updated_at,
         c.name as category_name, c.id as cat_id,
         MATCH(a.title, a.content) AGAINST(#{keyword} IN NATURAL LANGUAGE MODE) AS relevance
  FROM articles a
  LEFT JOIN categories c ON a.category_id = c.id
  WHERE a.status = 'PUBLISHED'
    AND MATCH(a.title, a.content) AGAINST(#{keyword} IN NATURAL LANGUAGE MODE)
  ORDER BY relevance DESC
  LIMIT #{limit} OFFSET #{offset}
</select>
```

**countSearchPublished** - Count for search pagination:

```xml
    AND MATCH(a.title, a.content) AGAINST(#{keyword} IN NATURAL LANGUAGE MODE)

```

**PublicArticleController.java** - REST endpoints (no auth required per SecurityConfig):
```java
@RestController
@RequestMapping("/api/articles")
public class PublicArticleController {

    @GetMapping
    public ResponseEntity<ArticleListResponse> getArticles(
            @RequestParam(defaultValue = "1") int page,
            @RequestParam(defaultValue = "10") int pageSize) {
        // Validate page >= 1, pageSize <= 100
        ArticleListRequest request = new ArticleListRequest(page, pageSize);
        return ResponseEntity.ok(publicArticleService.getArticles(request));
    }

    @GetMapping("/{id}")
    public ResponseEntity<PublicArticleResponse> getArticle(@PathVariable Long id) {
        return ResponseEntity.ok(publicArticleService.getArticleById(id));
    }

    @GetMapping("/category/{categorySlug}")
    public ResponseEntity<ArticleListResponse> getByCategory(
            @PathVariable String categorySlug,
            @RequestParam(defaultValue = "1") int page,
            @RequestParam(defaultValue = "10") int pageSize) {
        return ResponseEntity.ok(publicArticleService.getArticlesByCategory(categorySlug, page, pageSize));
    }

    @GetMapping("/tag/{tagSlug}")
    public ResponseEntity<ArticleListResponse> getByTag(
            @PathVariable String tagSlug,
            @RequestParam(defaultValue = "1") int page,
            @RequestParam(defaultValue = "10") int pageSize) {
        return ResponseEntity.ok(publicArticleService.getArticlesByTag(tagSlug, page, pageSize));
    }

    @GetMapping("/search")
    public ResponseEntity<ArticleListResponse> search(
            @RequestParam String q,
            @RequestParam(defaultValue = "1") int page,
            @RequestParam(defaultValue = "10") int pageSize) {
        return ResponseEntity.ok(publicArticleService.searchArticles(q, page, pageSize));
    }
}
```

Note: The controller does NOT require authentication (SecurityConfig has `anyRequest().permitAll()`).

Also update Article entity to include a `slug` field (not stored in DB, generated on-the-fly via generateSlug(title)).
